        // --- 1. ضع النص الصحيح هنا (بين علامتي التنصيص) ---
        const correctText = "الحمد لله الذي فضل بني آدم بالعلم والعمل على جميع العالم، والصلاة والسلام على محمد سيد العرب والعجم، وعلى آله وأصحابه ينابيع العلوم والحكم.
وبعد
فلما رأيت كثيرا من طلاب العلم في زماننا يجدون إلى العلم ولا يصلون [ومن منافعه وثمراته - وهي العمل به والنشر - يحرمون] (1) لما أنهم أخطأوا طريقه وتركوا شرائطه، وكل من أخطأ الطريق ضل، ولا ينال المقصود قل أو جل، فأردت وأحببت أن أبين لهم طريق التعلم على ما رأيت في الكتب وسمعت من أساتيذي أولي العلم والحكم، رجاء الدعاء لي من الراغبين فيه، المخلصين، بالفوز والخلاص في يوم الدين، بعد ما استخرت الله تعالى فيه، وسميته:
تعليم المتعلم طريق التعلم"; 

        // --- 2. ضع رابط جوجل شيت (الذي أنشأته سابقاً) هنا ---
        const scriptURL = "ضع_رابط_جوجل_هنا";

        // وظيفة التكبير
        function enableZoom() {
            const img = document.getElementById('displayImg');
            const container = document.getElementById('zoom-area');
            container.onmousemove = (e) => {
                const rect = container.getBoundingClientRect();
                const x = ((e.clientX - rect.left) / rect.width) * 100;
                const y = ((e.clientY - rect.top) / rect.height) * 100;
                img.style.transformOrigin = `${x}% ${y}%`;
                img.style.transform = "scale(2.5)";
            };
            container.onmouseleave = () => img.style.transform = "scale(1)";
        }
        window.onload = enableZoom;

        function normalize(t) { return t.trim().replace(/\s+/g, ' '); }

        function showResults() {
            const name = document.getElementById('studentName').value.trim();
            const id = document.getElementById('studentID').value.trim();
            if(!name || !id) return alert("يرجى إدخال اسمك ورقمك الجامعي أولاً!");

            const userText = document.getElementById('studentInput').value;
            const origWords = normalize(correctText).split(' ');
            const userWords = normalize(userText).split(' ');
            let correct = 0;
            let html = '';

            for (let i = 0; i < origWords.length; i++) {
                if (origWords[i] === userWords[i]) {
                    html += `<span class="word-correct">${origWords[i]}</span> `;
                    correct++;
                } else {
                    html += `<span class="word-wrong">${userWords[i] || "___"}</span> <span class="word-original">[الأصل: ${origWords[i]}]</span> `;
                }
            }

            const score = ((correct / origWords.length) * 100).toFixed(1) + "%";
            
            // إرسال البيانات لجوجل شيت
            fetch(scriptURL, { method: 'POST', mode: 'no-cors', body: JSON.stringify({name, id, score}) });

            document.getElementById('studentDataDisplay').innerHTML = `الطالب: ${name} | الرقم: ${id}`;
            document.getElementById('scoreDisplay').innerText = "نسبة الإتقان: " + score;
            document.getElementById('comparisonArea').innerHTML = html;
            document.getElementById('resultBox').style.display = 'block';
            document.getElementById('studentInput').disabled = true;
            document.getElementById('finishBtn').style.display = 'none';
        }
    </script>
